Usage Example

<!-- Include the theme -->
<link rel="stylesheet" href="@ignited-ui/themes/material.css">

<!-- Use the component -->
<ig-button variant="filled">Click Me</ig-button>

<!-- Listen to events -->
<script>
  const button = document.querySelector('ig-button');
  button.addEventListener('ig-click', (e) => {
    console.log('Button clicked!', e.detail);
  });
</script>
